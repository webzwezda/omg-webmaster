Функции

Функция - это блок кода, который можно многократно вызывать на исполнение. При этом функции можно передавать аргументы (параметры) с которыми она будет работать. Аргументы они же - параметры играют роль локальных переменных в теле функции.
Именна функций даются в стиле UpperCamelCase.

  1. Необязательные аргументы
  2. Массив аргументов функции
  3. Колличество аргументов функции
  4. Рекурсия
  5. CallBack Function

Необязательные аргументы

function print(msg, count) {
  // Необязательный параметр count.

    if (count == undefined) {
        count = 3;
    }

    for (var i = 0; i < count; i++) {
        document.write("<p>" + msg);
    }

    document.write("<hr/>");
}

// Вызов функций.

print("hello");
print("world", 7);

Массив аргументов функции

function max(a, b, c) {
//функция показывает максимальное значение аргумента, котороый передан при вызове


    var maxValue = Number.NEGATIVE_INFINITY;//присваеваем изначально минус бесконечность

    document.write("<p>Number.NEGATIVE_INFINITY = " + maxValue + "<hr/>");
    //перебираем циклом аргументы

    // arguments - свойство объекта max - который ефункцией

    for (var i = 0; i < arguments.length; i++)
        if (arguments[i] > maxValue)
            maxValue = arguments[i];

    return maxValue;
}

var res = max(-33, 33, 777, -666, -22, 888);

document.write("<p>Максимальное значение одного из аргументов функции = " + res);

Колличество аргументов функции

Функция, которая проверяет правильное ли количество аргументов передано. При вызове функции создается специальный масив содержащий в себе аргументы, которые были переданы при вызове. У этого массива есть свойство length, в котором хранится количество аргументов переданых функции.

function f(x, y, z) {

// Проверка, правильное ли количество аргументов передано.

if (arguments.length != 3) {
    document.write("<p><b style='color:red;'>ERROR:</b>
      <em>Требуется три аргумента при вызове Функция вызвана с "
        + arguments.length + параметрами!</em>");
}
else {
    document.write("<p><b style='color:green;'> Правильный вызов функции.</b> Аргументы: ");

    for (var i = 0; i < arguments.length; i++)
        document.write(arguments[i] + ", ");
    }
}

f(1, 2, 3, 4);   // Более трех аргументов - ошибка!

f(1, 2, 3);
f(1, 2)          // Менее трех аргументов - ошибка!

Рекурсия

function f(counter) {//функция принимает параметр counter


    counter--;//уменшается параметр на 1


    document.write("<p style='background-color:lightblue;'>Первая половина метода: " + counter);

    if (counter != 0)//проводим проверку если counter не равен 0

        f(counter);//вызываем функцию заново и уменьшаем counter еще раз

    //после того, как условие не выполняется выводится следующее сообщение

    document.write("<p style='background-color:yellow;'>Вторая половина метода: " + counter);
}

f(3);

CallBack Function

var add = function (a, b) {//переменная с функцией

    return a + b;           //выполняет сложение

};

var sub = function (a, b) {//переменная с функцией

    return a - b;           //выполняет вычитание

};

function show(callBackFunction, a, b) {

/*callBackFunction - обозначает,
что при вызове функции show
должны передать еще одну функцию, CallBack функцию.
Функция show сама решит, когда нузапустить CallBack Function*/

    var result = callBackFunction(a, b);
    document.write("<p>" + result);

};

show(add, 10, 20);//если передать в callBackFunction функцию add

show(sub, 50, 30);//если передать в callBackFunction функцию sub
