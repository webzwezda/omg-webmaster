While

Цикл while - выполняется до тех пор, пока условие удовлетворяет истиности

var str = "Угадайте задуманный цвет с пяти попыток.";

var maxAttempt = 5;   // Допустимое количество попыток.

var attempt = 0;      // Счетчик попыток.

var color = "red";    // Задуманный цвет.


while (attempt < maxAttempt) {
    attempt++;

    value = prompt(str + " Попытка " + attempt, "");

    if (value == null) {
        break;
    }

    if (value != color) {
        str = "Вы не угадали.";
        continue;
    }

    document.write("<p>Поздравляем, Вы угадали с " + attempt + " попытки!");
    break;
}

document.write("<p>Конец игры!");

Do While

Цикл с постустловием do-while - это цикл, в котором условие проверяется после выполнение тела цикла. Тело цикла выполняется хотя бы раз.

var input;

do {
    input = prompt("Введите некоторое значение.");
    document.write("<p>" + input);
}
while (input != null);

Цикл Дейкстры

При выполнении цикла Дейкстры в каждой итерации происходит вычисление охраняемых условий.

Цикл Дейкстры с одним охраняющим условием и одной охраняемой командой представляет собой, по сути, обычный цикл с предусловием (цикл «while»).

var a;

for (; ;) {

    a = prompt("Введите направление l/r/t/b");

    switch (a) {
        case 'l': // 'l' - охраняющее условие 1.

            {
                document.write("на лево <br />");    // охраняемая команда.

                continue;
            }
        case 'r': // 'r' - охраняющее условие 2.

            {
                document.write("на право <br />");   // охраняемая команда.

                continue;
            }
        case 't': // 'l' - охраняющее условие 1.

            {
                document.write("вверх <br />");    // охраняемая команда.

                continue;
            }
        case 'b': // 'r' - охраняющее условие 2.

            {
                document.write("вниз <br />");   // охраняемая команда.

                continue;
            }
        default:
            {
                exit = confirm("Выйти?");
                if (exit) {
                    document.write("Вы завершили выполнение.");
                    break;
                }
                else {
                    continue;
                }
            }
    }
    break;  // Прерывание цикла.

}
