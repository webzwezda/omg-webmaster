Тернарный оператор

Включает в себя три операнда - ( условие ? Если условие ИСТИННО : Если условие ЛОЖНО ).

Переменной max, присваиваем возвращаемое значение тернарного оператора.

var a = 1, b = 2, max = 0;

// ЕСЛИ:  (a > b)  ТО:  вернуть a  ИНАЧЕ: вернуть b;

max = (a > b) ? a : b;

document.write(max);

Знак “!”

Знак восклицания меняет логическое значение на противоположное.

var trueVal = true;
var falseVal = false;

if (trueVal) {
    document.write("Первый блок if выполнился");
}
else if (!falseVal) {
    document.write("Второй блок else if выполнился");
}
else {
    document.write("Ни один блок не выполнился");
}

If & Else

var a = prompt("Введите число 22", "0");

if (a == "22") {
  document.write("Введенное значение верно");
}
else {
  document.write("Введенное значение НЕ верно");
}

document.write(" <br/>Эта строка выполнится в любом случае");

Else if

var a = prompt("Введите слово Hello, или World", "Hello");

if (a == "Hello" || a == 'hello') {
    document.write("Введенное значение верно и равно Hello");
}
else if (a == "World" || a == 'world') {
   // Выполнится если выражение в перконструкции будет false

    document.write("Введенное значение верно и равно World");
}
else {
    document.write("Введенное значение НЕ верно");
}

document.write("<br/>Эта строка выполнится в любом случае");

Switch

var day = prompt("Введите номер дня недели: 1-2-3-4-5-6-7:");

switch (day) {

    case "1":
        {
            document.write("Понедельник");
        };
        break;
    case "2":
        {
            document.write("Вторник");
        };
        break;
    case "3":
        {
            document.write("Среда");
        };
        break;
    case "4":
        {
            document.write("Четверг");
        };
        break;
    case "5":
        {
            document.write("Пятница");
        };
        break;
    case "6":
        {
            document.write("Суббота");
        };
        break;
    case "7":
        {
            document.write("Воскресенье");
        };
        break;
    default:
        {
            document.write("Вы ввели несуществующий день недели");
        };
}
