Массивы

Массивы в JavaScript - это именованный набор не строго типизированный переменных, расположенных в памяти непосредственно друг за другом, доступ к которым осуществляется по индексу.

Свойство length

Массив - обьект, у которого имеется свойство length, это свойство хранит в себе длину массива.

Это свойство удаляет и/или добавляет елементы массива в зависимости от присвоения неотрицательного целого числа.

var array = ['h', 'e', 'l', 'l', 'o'];

array.length = 7;

for (var i = 0; i < array.length; i++) {
    document.write(array[i] + "<br />");
}

document.write("<hr />");

Ассоциативные массивы

Знак восклицания меняет логическое значение на противоположное.

var laptop = { cpu: "Core i7", ram: "4 GB", screen: "15" };

var key = prompt("Введите интересующий параметр (ram, cpu, screen)", "");

var value = laptop[key];
document.write(value + "<hr/>");

Двумерная матрица

Массивы бывают многомерными, для этого в каждый элемент массива следует расположить еще один массив. Таким образом возможно реализовать n-мерный массив.

var table = new Array(10);

for (var i = 0; i < table.length; i++) { // В таблице 10 строк

    table[i] = new Array(10);            // В каждой строке 10 столбцов

}

// Инициализация массива.

for (var row = 0; row < table.length; row++) {
    for (var col = 0; col < table[row].length; col++) {
        if ((row + col) % 2 == 1)
            table[row][col] = "1";
        else
            table[row][col] = "0";
        //document.write(table[row][col]); //вывод заполненой ячейки сразу

    }
    // document.write("<br/>");//разрыв строки

        }

//table[8][1] = "<img src='images/parquet3.jpg' width='50' height='50'/>";


for (var row = 0; row < table.length; row++) {
    for (var col = 0; col < table[row].length; col++) {
        document.write(table[row][col]);
    }
    document.write("<br/>");
}

var prod = table[8][1];
document.write("<hr/>" + prod);

Методы массива

Особенность метода join в том что он не обрабатывает последний элемент массива.

var arr = ["September", "October", "November",
   "December", "January", "Fabruary", "March",
    "April", "May", "June", "Jull", "August"];

document.write(arr);

document.write("<p><b>Значение которое возвращает метод - join(' | ');</b>");

var res = arr.join(" | ");
document.write("<p>" + res);

document.write("<p><b>Значение которое возвращает метод - join(< br/ >);</b>");

res = arr.join("<br/>");
document.write("<p>" + res);

Добавление в массив

concat() - создает и возвращает новый массив, содержащий элементы исходного массива, для которого был вызван метод concat(), и значения всех аргументов, переданных методу concat().

var a = [1, 2, 3];

var input = prompt("Введите новый элемент массива:", "");

while (input != null) {
    a = a.concat(input);
    input = prompt("Введите новый элемент массива:", "");
}

document.write(a.join("<br />"));

a.reverse()

Метод slice()

Метод splice() - универсальный метод для вставки и удаления элементов из массива, возвращает массив из удаленных элементов.

var a = [1, 2, 3, 4, 5, 6, 7];
document.write("<p>Исходный массив: " + a);

// УДАЛЕНИЕ.


/* splice (начальный индекс, сколько элементов удалить);
Если второй аргумент опущен, удаляются элементы массива до конца массива */

var d = a.splice(1, 3);   // Будут удалены: 2, 3, 4

document.write("<p>Удалены: " + d);
document.write("<p>Остались: " + a);

// ВСТАВКА.


/*(начальный индекс = 1, сколько элементов удалить = 0, значения для вставки с начальниндекса = 2, 3, 4); */
a.splice(1, 0, 2, 3, 4);
document.write("<p>После вставки: " + a);

Метод push()

push() - добавление одного или нескольких элементов в конец и массива возвращает его новую длину. (возвращаемое значение - количество элементов в новом массиве)

var stack = ["January", "February", "March"];
var lenght = stack.push("April");
document.write("<p>После добавления: " + stack.join(" "));
document.write("<p>Размер массива: " + lenght);

Метод pop()

pop() - выполняет удаление последнего элемента массива, уменьшает длину массива. (возвращаемое значение - удаленный элемент массива)

var stack = ["January", "February", "March", "April", "May", "June"];
document.write("<p>Исходный: " + stack.join(" "));
document.write("<p>Размер массива: " + stack.length);

document.write("<hr/>");

var lenght = stack.pop();
document.write("<p>После pop: " + stack.join(" "));
document.write("<p>Удаленный элемент: " + lenght);
document.write("<p>Размер массива после удаления: " + stack.length);

Метод shift()

shift() - удаляет и возвращает первый элемент массива смещая все последующие элементы на одну позицию вниз, чтобы занять место, освободившееся в начале массива.

var arr = [1, 2, 3];

document.write("<p>Исходный массив: " + arr.join(" "));
document.write("<hr/>");

var value = arr.shift();

document.write("<p>Массив после сдвига: " + arr.join(" "));
document.write("<p>Выдвинутое значение: " + value);

Метод unshift()

unshift() - добавляет элемент в начало массива. Смещает существующие элементы в сторону больших индексов для освобождения места, добавляет элемент или элементы в начало массива и возвращает новую длину массива.

var arr = [1, 2, 3];
document.write("<p>Исходный массив: " + arr.join(" "));

document.write("<hr/>");

arr.unshift(0);
document.write("<p>Массив после сдвига: " + arr.join(" "));

document.write("<hr/>");

arr.unshift(-3, -2, -1);
document.write("<p>Массив после сдвига: " + arr.join(" "));
